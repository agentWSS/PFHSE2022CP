def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """Нахождение корней $f(x) = 0$ через итерации Ньютона.
    
    Parameters
    ----------
    f : callable
        Функция, корни которой мы хотим найти.
    fder : callable
        Производная `f`.
    x0 : float
        Начальное приближение итераций Ньютона.
    eps : float
        Заданная точность. 
        Алгоритм прекращает работу когда расстояние между последовательными приближениями меньше `eps`.
        По умолчанию 1e-5.
    maxiter : int
        Максимальное число итераций (по умолчанию 100).
        Алгоритм прекращается, когда число итераций достигает `maxiter`.
        Этот параметр нужен лишь для предотвращения бесконечного зацикливания.
    
    Returns
    -------
    x : float
        Найденное приближение к корню.
    niter : int
        Количество итераций.
    """
    x_np1 = x0
    x_n = x0 + 2 * eps
    i = 0

    while abs(x_np1 - x_n) >= eps and i < maxiter:
        x_n = x_np1
        x_np1 = x_n - f(x_n) / fder(x_n)

        i += 1
    x, niter = float(x_np1), i
    return x, niter

def mod_newton(f, fder, x0, m, eps=1e-5, maxiter=100):
    """Нахождение корней $f(x) = 0$ через итерации Ньютона.
    
    Parameters
    ----------
    f : callable
        Функция, корни которой мы хотим найти.
    fder : callable
        Производная `f`.
    x0 : float
        Начальное приближение итераций Ньютона.
    eps : float
        Заданная точность. 
        Алгоритм прекращает работу когда расстояние между последовательными приближениями меньше `eps`.
        По умолчанию 1e-5.
    maxiter : int
        Максимальное число итераций (по умолчанию 100).
        Алгоритм прекращается, когда число итераций достигает `maxiter`.
        Этот параметр нужен лишь для преcдотвращения бесконечного зацикливания.
    
    Returns
    -------
    x : float
        Найденное приближение к корню.
    niter : int
        Количество итераций.
    """
    x_np1 = x0
    x_n = x0 + 2 * eps
    i = 0

        i += 1
    x, it = float(x_np1), i
    return x, it

def complex_nwet(f, fder, z0, eps=1e-5, maxiter=100):
    x_np1 = z0
    x_n = z0 + 2 * complex(eps, eps)
    i = 0

    while abs(x_np1 - x_n) >= eps and i < maxiter:
        x_n = x_np1
        x_np1 = x_n - f(x_n) / fder(x_n)

        i += 1
    return x_np1, i

def newton_system(F, jcb, X0, eps=1e-5, maxiter=100):
    """Нахождение корней $f(x) = 0$ через итерации Ньютона.
    
    Parameters
    ----------
    F : callable
        Вектор-функция системы, которую мы хотим решить.
    jcb : callable
        Якобиан `f`.
    X0 : array-like of floats, shape (n,)
        Начальное приближение итераций Ньютона.
    eps : float
        Заданная точность. 
        Алгоритм прекращает работу когда расстояние между последовательными приближениями меньше `eps`.
        По умолчанию 1e-5.
    maxiter : int
        Максимальное число итераций (по умолчанию 100).
        Алгоритм прекращается, когда число итераций достигает `maxiter`.
        Этот параметр нужен лишь для предотвращения бесконечного зацикливания.
    
    Returns
    -------
    X : array-like of floats, shape (n,)
        Найденное приближение к корню.
    niter : int
        Количество итераций.
    """
    x_np1 = X0
    x_n = X0 + 2 * eps
    i = 0

    while np.linalg.norm(x_np1 - x_n) >= eps and i < maxiter:
        x_n = x_np1
        dx = solve(jcb(x_n), -F(x_n))
        x_np1 = x_n + dx

        i += 1
    X, niter = x_np1, i
    return X, niter

def func2(X):
    x, y, z = X
    return np.array([x**2 + y**2 + z**2 - 25,
                     x*y + y*z + z*x - 5,
                     x + y - 3])

def jac2(X):
    r"""Return the Jacobian of `func2(X)`.
    
    
    The Jacobian matrix is defined as 
    
    $$
    J_{ij} = \partial f_i / \partial x_j
    $$
    
    so that the first row contains the derivatives of $f_0$
    with respect to the first, second etc coordinates; the second
    row contains the derivatives of $f_1$ with respect to
    the first, second etc coordinates; and so on.
    """
    x_v, y_v, z_v = X

    x, y, z = sp.symbols('x y z')
    f1, f2, f3 = func2([x, y, z])
    F = sp.Matrix([f1, f2, f3])
    
    J = F.jacobian([x, y ,z])
    # J_v = J.subs([(x, x_v), (y, y_v), (z, z_v)])
    
    s = (x, y, z)
    J_func = sp.lambdify(s, J, modules='numpy')
    J_ev = J_func(x_v, y_v, z_v)
    
    return J_ev
